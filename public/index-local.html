		response.redirect("/hello/senchacon");
	});
	
	app.get("/hello/senchacon", function(request, response) {
		response.send("Hello SenchaCon!");
	});
	
	app.listen(4005);

	function loadUser(req, res, next) {
		db.getUser(req.params.id, function(err, user) {
			if (user) {
				req.user = user;
    			next();
  			} else 
  				next(new Error('Failed to load user ' + req.params.id));
		});
	}

	function restrictToSelf(req, res, next) {
  		(req.authenticatedUser.id == req.user.id) ? next() : 
  			next(new Error('Unauthorized'));
	}

	function restrictToRole(role) {
  		return function(req, res, next) {
    		(req.authenticatedUser.role == role) ? next() : 
    			next(new Error('Unauthorized'));
  		}
	}

	app.get("/user/:id/edit", loadUser, restrictToSelf, function(req, res) {
  		res.send("Editing user " + req.user.name);
	});
	
	app.get("/user/:id/destroy", loadUser, restrictToRole("admin"), function(req, res) {
  		res.send('Deleting user ' + req.user.name);
	});
